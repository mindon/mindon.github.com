package com.mindon.util;

import java.util.PropertyResourceBundle;

import java.util.ResourceBundle;

public class ConfProperties

{

private static final String confName = &quot;config&quot;;

/**

  • 根据key返回一个字符串的值

  • @param key

*/

public static String get(String key) {

return get(confName, key);

}

/**

  • 根据key返回一个字符串的值

  • @param conf

  • @param key

*/

public static String get(String conf, String key) {

ResourceBundle resBundle = PropertyResourceBundle.getBundle(conf);

try {

if(null != resBundle) {

return resBundle.getString(key);

} else {

System.out.println(&quot;[-FAIL-&gt;&gt;&quot;+conf+&quot;.properties not found&quot;);

return &quot;&quot;;

}

} catch(java.util.MissingResourceException e) {

System.out.println(&quot;[-FAIL-&gt;&gt;&quot; +key +&quot; property not found&quot;);

return &quot;&quot;;

}

}

}
