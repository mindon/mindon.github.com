file lake0.m: x - y distribution

%lake0(m,num)

%m decides how smooth the curve you want to be.

%num decides how many countors you want to show.

%example:lake0(160,20)

function lake0(m,num)

%all known parameters setup

n=6;

%u0=3.854167;

%c0=0.599653;

u0=0;

c0=7.195833;

c=[-0.121111 -0.000089 -0.233839 -0.114921 0.015436 -0.368833];

d=[0.033955 -0.034494 -0.017328 -0.086451 0.073220 0.000013];

r=0:1/m:1.0;

thita=0:2pi/m:2pi;

%Initalize the matrixs for higher speed

rp=(r(ones(size(thita)),:))‘;

thita=thita(ones(size®),:);

%clear useless vetor

clear r;

x=31rp.cos(thita);

y=31rp.sin(thita);

u=rp.(c(1)cos(thita)+d(1)*sin(thita));

for k=2:n

u=u+(rp.^k).(c(k)cos(kthita)+d(k)sin(k*thita));

end

%clear useless vetor

clear c d k n m;

clear rp thita;

u=u0+c0/2+u;%Final result

%define the color map

colormap(zeros(64,3));

grid off

axis equal,hold;

axis([-35 35 -35 35]);

set(gcf,‘Color’,[1 1 1]);

set(gca,‘Box’,‘on’,‘LineWidth’,1.5);

rAngle=pi;

xr=xcos(rAngle)+ysin(rAngle);

yr=-xsin(rAngle)+ycos(rAngle);

%[C,h]=contour(x,y,u,num);

[C,h]=contour(xr,yr,u,num);

%clear useless vetor

clear x y u;

title(‘湖水内部环境污染综合指标分布’);

xlabel(‘x (m)’);

ylabel(‘y (m)’);

%clabel©;

%clabel(C,h);

clabel(C,‘manual’);

hold off

%lake1(m,num)

%m decides how smooth the curve you want to be.

%num decides how many countors you want to show.

%example:lake1(160,20)

function lake(m,num)

n=6;

%u0=3.854167;

%c0=0.599653;

u0=0;

c0=7.195833;

c=[-0.121111 -0.000089 -0.233839 -0.114921 0.015436 -0.368833];

d=[0.033955 -0.034494 -0.017328 -0.086451 0.073220 0.000013];

r=0:31/m:31.0;

thita=0:360/m:360;

rp=(r(ones(size(thita)),:))‘;

thita=thita(ones(size®),:);

u=(rp/31).(c(1)cos(thitapi/180)+d(1)sin(thita*pi/180));

for k=2:n

u=u+((rp/31).^k).(c(k)cos(k(thitapi/180))+d(k)sin(k(thita*pi/180)));

end

u=u0+c0/2+u;

colormap(zeros(64,3));

[C,h]=contour(thita,rp,u,num);

set(gcf,‘Color’,[1 1 1]);

set(gca,‘Box’,‘on’,‘LineWidth’,1.5);

set(h,‘LineWidth’,1.2);

title(‘湖水内部环境污染综合指标分布’);

ylabel(‘半径 r (m)’);

xlabel(‘角度 (度)’);

%clabel(C,h);

clabel(C,‘manual’);

hold off

2005的开始

Jan 4th, 2005 8:51 am

BlogMS original blog key: 1000139151, blog id: airoff History stat: 浏览/评论：97/1 , 日期：2005年1月4日 08:51

1月2日，在待屋里编了一整天程序后，决定出门做点什么。想起了在何香凝美术馆的毕加索版画展。因为美术馆离住的地方不外两站路，于是就走了过去，顺便给自己点活动量。票价35块，进去的时候还经过了如机场安检一般的程序，保安看到我带了相机还特意提醒不允许拍照——有点失望。美术馆里同时有另外一个画家（刘贵宾）的油画展，就是圣诞节在美术馆外拍照片时看到在招贴底下留影的那位。

看版画展的人不少，还有很多小朋友随他们的老师来学习的。我也不懂什么版画，只是凭自己的感觉和知道的一丁点有限的了解来理解自己眼前的线条和画面。我想，看展览不外是两种目的：企图理解艺术家所要表达的内容和找寻自己期待的内容。我的知识和理解能力大概对艺术家原本的表达意图揣测是十分有限的，大多时候我只是在找寻自己所期待的线条和意境，至于符不符合作者最初想表达的东西，我并不是太在意。
